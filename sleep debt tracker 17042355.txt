    function scrollToSection(id) { document.getElementById(id).scrollIntoView({behavior:'smooth'}); }
    function calculateDebt() {
      const r = parseFloat(document.getElementById('required').value);
      const a = parseFloat(document.getElementById('actual').value);
      const d = parseInt(document.getElementById('days').value);
      const res = document.getElementById('result');
      const recList = document.getElementById('recovery-list');
      const refList = document.getElementById('references-list');
      if(isNaN(r) || isNaN(a) || isNaN(d)) {
        res.innerHTML = '<p style="color:salmon;">Enter valid numbers.</p>';
        recList.innerHTML = '<li>Please calculate to see tips.</li>';
        refList.innerHTML = '<li>Calculate to view references.</li>';
        return;
      }
      const debt = (r - a) * d;
      let html = '<div class="result-details">';
      let recHtml = '';
      let refHtml = '';
      if(debt > 0) {
        html += `<p><strong>Your sleep debt: ${debt} hrs.</strong></p><ul><li>Prioritize earlier bedtime.</li><li>Short restorative naps.</li><li>Stick to a schedule.</li><li>Optimize environment.</li></ul>`;
        recHtml = `<li>Increase nightly sleep by 15-30 mins.</li><li>Maintain consistent sleep schedule.</li><li>Limit blue light before bed.</li><li>Create a cool, dark sleep environment.</li><li>Consider short naps.</li>`;
        refHtml = `<li>Van Dongen et al., Dose-response effects of sleep loss (2003).</li><li>National Sleep Foundation guidelines on sleep debt.</li><li>Hirshkowitz et al., National Sleep Foundation recommendations (2015).</li>`;
      } else if(debt < 0) {
        html += `<p><strong>Sleep surplus: ${Math.abs(debt)} hrs.</strong></p><ul><li>Avoid oversleeping habits.</li><li>Set regular wake time.</li><li>Limit daytime naps.</li><li>Balance daily activity.</li></ul>`;
        recHtml = `<li>Avoid sleeping excessively.</li><li>Get morning sunlight exposure.</li><li>Exercise earlier in day.</li><li>Keep consistent wake time.</li><li>Limit naps to 20 mins.</li>`;
        refHtml = `<li>Kim et al., Health effects of oversleeping (2013).</li><li>American Academy of Sleep Medicine oversleeping study.</li><li>Tan et al., Optimal sleep duration review (2016).</li>`;
      } else {
        html += `<p><strong>Perfect sleep achieved!</strong></p><ul><li>Keep this routine.</li><li>Monitor sleep quality.</li><li>Maintain healthy habits.</li><li>Track monthly for consistency.</li></ul>`;
        recHtml = `<li>Your sleep is balanced.</li><li>Continue consistent bedtime.</li><li>Monitor for changes.</li><li>Maintain good sleep hygiene.</li><li>Use tracker monthly.</li>`;
        refHtml = `<li>Centers for Disease Control and Prevention sleep guidelines.</li><li>Watson et al., Sleep health framework (2015).</li><li>Buysse, Sleep health definition (2014).</li>`;
      }
      html += '</div>';
      res.innerHTML = html;
      recList.innerHTML = recHtml;
      refList.innerHTML = refHtml;
    }
    function toggleFooter(id) { ['privacy','disclaimer','contact'].forEach(sec => { document.getElementById(sec).style.display = sec===id ? 'block' : 'none'; }); }
  </script>
